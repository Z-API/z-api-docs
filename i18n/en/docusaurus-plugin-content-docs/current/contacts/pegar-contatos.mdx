# Get contacts

Get the list of WhatsApp contacts from your instance.

## Endpoint

```http
GET https://api.z-api.io/instances/SUA_INSTANCIA/token/SEU_TOKEN/contacts
```

### Headers

- Client-Token: string (required)

## Responses

### 200 OK

```json
[
  {
    "name": "Nome e sobrenome do contato",
    "short": "Nome do contato",
    "notify": "Nome no WhatsApp",
    "vname": "Nome no vcard",
    "phone": "5599999999999"
  }
]
```

### Contact Fields

| Attributes | Type   | Description                                                                                         |
| --------- | ------ | ------------------------------------------------------------------------------------------------- |
| phone     | string | Phone of the contact                                                                               |
| name      | string | Full name of the contact, will only return filled if you have the number in your contacts          |
| short     | string | Name of the contact, will only return filled if you have the number in your contacts               |
| vname     | string | Name of the contact if you have it as a contact                                                  |
| notify    | string | Name informed in WhatsApp settings                                                                 |

### Query Parameters

Insert attributes of `page` and `pageSize`, it is possible to insert in the URL of the request.

#### Required

| Attributes | Type    | Description                                                                        |
| --------- | ------- | ---------------------------------------------------------------------------------- |
| page      | integer | Used for pagination; specify here the chat page you want to search               |
| pageSize  | integer | Specifies the size of the chat return per page                                  |

**Example:**

```
{{BASE_URL}}/instances/{INSTANCE_ID}/token/{INSTANCE_TOKEN}/chats?page=1&pageSize=20
```

### Common Errors

| Code | Reason               | How to resolve                              |
| ---- | -------------------- | ------------------------------------------- |
| 401  | Invalid token        | Check the header `Client-Token`          |
| 404  | Instance does not exist | Confirm the `instanceId`                    |
| 5xx  | Internal error       | Try again; open support if it persists     |

// ⚠️ SEGURANÇA: Use variáveis de ambiente para credenciais

    return value ? value : (char*)defaultValue;

char* instanceId = getEnv("ZAPI_INSTANCE_ID", "SUA_INSTANCIA");
char* instanceToken = getEnv("ZAPI_INSTANCE_TOKEN", "SEU_TOKEN");
char* clientToken = getEnv("ZAPI_CLIENT_TOKEN", "seu-token-de-seguranca");

struct MemoryStruct {
    char* memory;
    size_t size;
};

static size_t WriteMemoryCallback(void* contents, size_t size, size_t nmemb, void* userp) {
    size_t realsize = size * nmemb;
    struct MemoryStruct* mem = (struct MemoryStruct*)userp;

    char* ptr = realloc(mem->memory, mem->size + realsize + 1);
    if (!ptr) {
        return 0;
    }

    mem->memory = ptr;
    memcpy(&(mem->memory[mem->size]), contents, realsize);
    mem->size += realsize;
    mem->memory[mem->size] = 0;

    return realsize;
}

int main() {
    // ⚠️ SEGURANÇA: Sempre use HTTPS
    CURL* curl = curl_easy_init();
    if (curl) {
        char url[512];
        snprintf(url, sizeof(url), "https://api.z-api.io/instances/%s/token/%s/contacts",
                 instanceId, instanceToken);

        struct curl_slist* headers = NULL;
        char tokenHeader[256];
        snprintf(tokenHeader, sizeof(tokenHeader), "Client-Token: %s", clientToken);
        headers = curl_slist_append(headers, tokenHeader);

        struct MemoryStruct chunk;
        chunk.memory = malloc(1);
        chunk.size = 0;

        curl_easy_setopt(curl, CURLOPT_URL, url);
        curl_easy_setopt(curl, CURLOPT_HTTPHEADER, headers);
        curl_easy_setopt(curl, CURLOPT_WRITEFUNCTION, WriteMemoryCallback);
        curl_easy_setopt(curl, CURLOPT_WRITEDATA, (void*)&chunk);
        curl_easy_setopt(curl, CURLOPT_TIMEOUT, 30L);

        CURLcode res = curl_easy_perform(curl);
        long responseCode;
        curl_easy_getinfo(curl, CURLINFO_RESPONSE_CODE, &responseCode);

        if (res == CURLE_OK && responseCode >= 200 && responseCode < 300) {
            // ⚠️ SEGURANÇA: Não logue tokens ou dados sensíveis
            printf("Contatos obtidos com sucesso\n");
        } else {
            // ⚠️ SEGURANÇA: Não exponha detalhes sensíveis em logs
            fprintf(stderr, "Erro HTTP %ld: Requisição falhou\n", responseCode);
        }

        free(chunk.memory);
        curl_slist_free_all(headers);
        curl_easy_cleanup(curl);
    }

    return 0;
}
